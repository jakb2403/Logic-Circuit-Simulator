DEVICES
    SW1, SW2, SW3, SW4 = SWITCH(1);
    CLK1 = CLOCK(20);
    OR1 = OR(2);
    NAND1 = NAND(2);
    NOR1 = NOR(2);
    XOR1 = XOR(2);
    DTYPE1 = DTYPE;
END 

CONNECT
    SW1 -> NAND1.I1;
    SW2 -> OR1.I2;
    SW3 -> NOR1.I2;
    SW4 -> DTYPE1.DATA;
    CLK1 -> OR1.I1, NAND1.I2;
    NAND1 -> DTYPE1.CLK;
    OR1 -> NOR1.I1;
    DTYPE.QBAR -> XOR1.I1;
    NOR1 -> XOR1.I2;
END 

MONITOR
    DTYPE1.Q;
    XOR1;
    NAND1;
END
