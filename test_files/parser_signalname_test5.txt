DEVICES
    SW1, SW2, SW3 = SWITCH(1);
    CLK1 = CLOCK(20);
    AND1, AND2 = AND(2);
    OR1, OR2 = OR(2);
    NAND1 = NAND(2);
    NOR1 = NOR(2);
    XOR1 = XOR;
    DTYPE1 = DTYPE;
END

CONNECT
    SW1 -> AND1.DATA;
    SW2 -> AND1.I2, NAND1.I1;
    SW3 -> NAND1.I2, DTYPE1.DATA;
    CLK1 -> DTYPE1.CLK, NOR1.I1, OR1.I1, XOR1.I1;
    AND1 -> NOR1.I2;
    NAND1 -> OR1.I2;
    DTYPE1.Q -> XOR1.I2;
    NOR1 -> AND2.I1;
    OR1 -> AND2.I2;
    XOR1 -> OR2.I1;
    DYPE1.QBAR -> OR2.I2;
END

MONITOR
    AND2;
    OR2;
    NOR1.I1;
END
